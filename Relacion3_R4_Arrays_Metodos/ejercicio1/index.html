                    • zip two arrays together
                    
                    function zipArrays(arr1 = [], arr2 = []) {
                        const length = Math.min(arr1.length, arr2.length);
                        const zipped = [];
                    
                        for (let i = 0; i < length; i++) {
                            zipped.push([arr1[i], arr2[i]]);
                        }
                    
                        return zipped;
                    }
                    
                    const arr1 = [1, 2, 3];
                    const arr2 = [4, 5, 6];
                    document.write(zipArrays(arr1, arr2));
